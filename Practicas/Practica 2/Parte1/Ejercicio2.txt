;Ejercicio 2b-
PA EQU 30H
CA EQU 32H

ORG 1000h
MSJ1 DB "Llave prendida"
MSJ2 DB "Llave apagada"

ORG 2000h
MOV AL, 0FFH
OUT CA, AL
IN AL, PA
AND AL, 10000000b
JZ PRENDIDA 
MOV AL, 13
MOV BX, OFFSET MSJ2
INT 7
PRENDIDA:
MOV AL, 14
MOV BX, OFFSET MSJ1
INT 7
HLT 
END

CB EQU 33H

OUT CB, AL

OUT CA, AL

END
