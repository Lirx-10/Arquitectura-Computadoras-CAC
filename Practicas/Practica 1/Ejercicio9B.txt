ORG 1000H
NUM DB 5
N DB 2
ORG 3000H
ROTAR_IZQ:
  PUSH AX
  MOV AH, [BX]
  ADD AH, AH
  JNC NO_CARRY
  INC AH
  NO_CARRY:
  MOV [BX],AH
  POP AX
RET

ROTAR_IZQ_N:
  PUSH AX
  LOOP:
  CMP AL, 0
  JZ FIN
  CALL ROTAR_IZQ
  DEC AL
  JMP LOOP
  FIN:
  POP AX
RET

ORG 2000H
MOV BX, OFFSET NUM
MOV AL, N
CALL ROTAR_IZQ_N
HLT
END
