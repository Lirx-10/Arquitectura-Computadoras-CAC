ORG 1000H
CADENA DB "hOla",00H
RESULTADO DB ?

ORG 3000H
CONTAR_MIN:
  PUSH BX
  MOV DL, 0
  LOOP:
    CMP BYTE PTR [BX], 0
    JZ FIN
    CMP BYTE PTR [BX], 'a'
    JS MAYUSCULA
    CMP BYTE PTR [BX], 'z'
    JNS MAYUSCULA
    INC DL
    MAYUSCULA:
    INC BX
    JMP LOOP
  FIN:
  POP BX
RET

ORG 2000H
MOV BX, OFFSET CADENA
CALL CONTAR_MIN
HLT
END
